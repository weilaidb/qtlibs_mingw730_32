☆ API名称
FFmpeg/macros/FFALIGN

☆ 简介
#define FFALIGN(x, a) (((x)+(a)-1)&~((a)-1))
求x按a对齐的边界值


☆ 参数


☆ 返回值


☆ 错误码



☆ 示例
int stride = FFALIGN(width, 8);

#include <stdio.h>

#define FFALIGN(x, a) (((x)+(a)-1)&~((a)-1))



int main()
{
    printf("FFALIGN(0,16):%u\n", FFALIGN(0,16));
    printf("FFALIGN(1,16):%u\n", FFALIGN(1,16));
    printf("FFALIGN(10,16):%u\n", FFALIGN(10,16));
    printf("FFALIGN(17,16):%u\n", FFALIGN(17,16));
    printf("FFALIGN(20,16):%u\n", FFALIGN(20,16));
    printf("FFALIGN(31,16):%u\n", FFALIGN(31,16));
    printf("FFALIGN(32,16):%u\n", FFALIGN(32,16));
    printf("FFALIGN(33,16):%u\n", FFALIGN(33,16));


    return 0;
}
