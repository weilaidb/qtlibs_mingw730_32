        window.onload = function() {
            var Go = document.getElementById("a1");
            var NoGo = document.getElementById("a2");

            function permitAccess(){
                alert("Go right ahead -- have fun!");
            }

            function refuseAccess(e){
                alert("sorry, but you are not yet authorized to follow this link!");
                e.preventDefault();
            }

            Go.addEventListener('click', permitAccess);
            NoGo.addEventListener('click', refuseAccess);
        }
