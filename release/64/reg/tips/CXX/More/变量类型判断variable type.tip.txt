#include<typeinfo>
#include<iostream>
using namespace std;

int testpara(int a, char *p)
{
    cout << typeid(a).name() << "\n";
    cout << typeid(p).name() << "\n";	
	return 0;
}
 
int main()
{
    int i;
    char *chr;
    const char *cchr;
    cout << typeid(i).name() << "\n";
    cout << typeid(chr).name() << "\n";
    cout << typeid(cchr).name() << "\n";
    //输出结果为i(表示int),即类型的名称首字母 
	
	testpara(10,10);
    return 0;
}
