package com.weilaidb.ajax.controller;

import com.weilaidb.ajax.entity.SearchCriteria;
import com.weilaidb.ajax.entity.User;
import com.weilaidb.ajax.pojo.AjaxResponseBody;
import com.weilaidb.ajax.service.UserService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.validation.Errors;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RestController;

import javax.validation.Valid;
import java.util.List;
import java.util.stream.Collectors;

@RestController
public class SearchController {

    UserService userService;

    @Autowired
    public void setUserService(UserService userService) {
        this.userService = userService;
    }

    @PostMapping("/api/search")
    public ResponseEntity<?> getSearchResultViaAjax(@Valid @RequestBody SearchCriteria search, Errors errors) {

        AjaxResponseBody result = new AjaxResponseBody();

        //If error, just return a 400 bad request, along with the error message
        if (errors.hasErrors()) {

            result.setMsg(errors.getAllErrors().stream().map(x -> x.getDefaultMessage()).collect(Collectors.joining(",")));
            return ResponseEntity.badRequest().body(result);

        }

        List<User> users = userService.findByUserNameOrEmail(search.getUsername());
        if (users.isEmpty()) {
            result.setMsg("no user found!" + search.getUsername() +",mail:" + search.getEmail());
            //示例多内容
            users.add(new User(search.getUsername(),search.getPassword(),search.getEmail()));
            users.add(new User("nihaoma2","hahha","ex"));
            users.add(new User("nihaoma3","hahha",search.getEmail()));
        } else {
            result.setMsg("success");
        }
        result.setResult(users);

        //多行数据
        return ResponseEntity.ok(result);
        //一行数据
//        return ResponseEntity.badRequest().body(result);
    }
}
