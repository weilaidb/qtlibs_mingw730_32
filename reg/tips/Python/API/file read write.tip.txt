# -*- coding: utf-8 -*-

def fileread(filename):
    with open(filename, 'r') as f:
        # print(f.read())
        result = f.read()
        f.close()
        return result


def filewrite(filename, content):
    with open(filename, 'w') as f:
        # print(f.read())
        f.write(content)
        f.close()
        return 0
